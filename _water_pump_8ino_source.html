192 void EEPROMWritelong(int address, long value)
193 {
194  //Decomposition from a long to 4 bytes by using bitshift.
195  //One = Most significant -> Four = Least significant byte
196  byte four = (value & 0xFF);
197  byte three = ((value >> 8) & 0xFF);
198  byte two = ((value >> 16) & 0xFF);
199  byte one = ((value >> 24) & 0xFF);
200 
201  //Write the 4 bytes into the eeprom memory.
202  EEPROM.write(address, four);
203  EEPROM.write(address + 1, three);
204  EEPROM.write(address + 2, two);
205  EEPROM.write(address + 3, one);
206 }
