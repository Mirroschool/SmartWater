212 void EEPROMWritelong(int address, long value)
213 {
214  // Decomposition from a long to 4 bytes by using bitshift.
215  // One = Most significant -> Four = Least significant byte
216  byte four = (value & 0xFF);
217  byte three = ((value >> 8) & 0xFF);
218  byte two = ((value >> 16) & 0xFF);
219  byte one = ((value >> 24) & 0xFF);
220 
221  // Write the 4 bytes into the EEPROM memory.
222  EEPROM.write(address, four);
223  EEPROM.write(address + 1, three);
224  EEPROM.write(address + 2, two);
225  EEPROM.write(address + 3, one);
226 }

231 long EEPROMReadlong(long address)
232 {
233  // Read the 4 bytes from the EEPROM memory.
234  long four = EEPROM.read(address);

238 
239  // Return the recomposed long by using bitshift.
240  return ((four << 0) & 0xFF) + ((three << 8) & 0xFFFF) + ((two << 16) & 0xFFFFFF) + ((one << 24) & 0xFFFFFFFF);
